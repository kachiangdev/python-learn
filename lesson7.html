🏁 Step 1: Welcome & Character Creation

# 🎭 PYTHON ADVENTURE GENERATOR 🎭 print("=" * 50) print("🌟 WELCOME TO THE PYTHON ADVENTURE GENERATOR! 🌟") print("=" * 50) print() # Get player information (Lesson 5: Input) print("Let's create your character!") name = input("🧙‍♂️ What's your character's name? ") age = int(input("🎂 How old are you? ")) favorite_color = input("🎨 What's your favorite color? ") # String manipulation (Lesson 4) character_name = name.title() # Capitalize first letter print(f"\n✨ Welcome, {character_name} the {favorite_color.upper()} adventurer! ✨")

🎲 Step 2: Character Stats Calculator

# Calculate character stats (Lesson 3: Arithmetic + Lesson 2: Variables) print("\n" + "=" * 30) print("📊 CALCULATING YOUR STATS...") print("=" * 30) # Base stats using arithmetic operations strength = (age * 2) + 10 magic_power = (len(character_name) * 5) + 15 luck = (age % 10) + (len(favorite_color) * 2) total_power = strength + magic_power + luck # Display stats with formatting (Lesson 1: Printing) print(f"💪 Strength: {strength}") print(f"🔮 Magic Power: {magic_power}") print(f"🍀 Luck: {luck}") print(f"⚡ TOTAL POWER: {total_power}") # Character type based on highest stat (Lesson 6: Conditions) if strength > magic_power and strength > luck: character_type = "WARRIOR" special_ability = "Mighty Strike" elif magic_power > strength and magic_power > luck: character_type = "WIZARD" special_ability = "Arcane Blast" else: character_type = "ROGUE" special_ability = "Lucky Strike" print(f"\n🎭 You are a {character_type}!") print(f"✨ Special Ability: {special_ability}")

⚔️ Step 3: Adventure Scenario Generator

# Adventure choice (Lesson 5: Input + Lesson 6: Conditions) print("\n" + "=" * 35) print("🗺️ CHOOSE YOUR ADVENTURE!") print("=" * 35) print("1. 🏰 Explore the Mysterious Castle") print("2. 🌲 Venture into the Enchanted Forest") print("3. 🏔️ Climb the Dragon's Mountain") choice = input("\nEnter your choice (1, 2, or 3): ") # Different adventures based on choice if choice == "1": location = "Mysterious Castle" monster = "Stone Guardian" treasure_base = 100 danger_level = "Medium" elif choice == "2": location = "Enchanted Forest" monster = "Forest Troll" treasure_base = 75 danger_level = "Low" elif choice == "3": location = "Dragon's Mountain" monster = "Ancient Dragon" treasure_base = 200 danger_level = "High" else: location = "Random Portal" monster = "Mysterious Shadow" treasure_base = 50 danger_level = "Unknown" print(f"\n🎯 Adventure: {location}") print(f"👹 You encounter a {monster}!") print(f"⚠️ Danger Level: {danger_level}")

💰 Step 4: Battle & Treasure Calculator

# Battle outcome (Lesson 6: Conditions + Lesson 3: Math) print("\n" + "=" * 25) print("⚔️ BATTLE TIME!") print("=" * 25) # Calculate success chance based on stats success_chance = total_power + (luck * 2) required_power = treasure_base + 50 if success_chance >= required_power: battle_result = "VICTORY" treasure_multiplier = 2.0 experience_gained = 100 print("🎉 VICTORY! You defeated the monster!") elif success_chance >= (required_power * 0.7): battle_result = "NARROW VICTORY" treasure_multiplier = 1.5 experience_gained = 75 print("😅 NARROW VICTORY! It was a tough fight!") else: battle_result = "RETREAT" treasure_multiplier = 0.5 experience_gained = 25 print("😬 You had to retreat, but you learned something!") # Calculate rewards (Lesson 3: Arithmetic) gold_found = int(treasure_base * treasure_multiplier) gems_found = gold_found // 20 # Floor division total_treasure_value = gold_found + (gems_found * 15) print(f"\n💰 Gold Found: {gold_found}") print(f"💎 Gems Found: {gems_found}") print(f"💰 Total Treasure Value: {total_treasure_value}") print(f"⭐ Experience Gained: {experience_gained} XP")

📜 Step 5: Epic Story Generator

# Generate epic story (Lesson 4: String manipulation + f-strings) print("\n" + "=" * 40) print("📜 YOUR EPIC ADVENTURE STORY") print("=" * 40) # Create dynamic story with all our variables adventure_story = f""" Once upon a time, {character_name}, a {age}-year-old {character_type.lower()}, embarked on a legendary quest. Wearing {favorite_color.lower()} armor that shimmered in the sunlight, our hero ventured into the {location.lower()}. With {strength} strength, {magic_power} magic power, and {luck} luck points, {character_name} was ready for anything! The journey led to an encounter with the fearsome {monster.upper()}! Using the powerful "{special_ability}" technique, our brave {character_type.lower()} achieved a {battle_result.lower()}! The adventure yielded {gold_found} gold pieces and {gems_found} precious gems. {character_name}'s power level of {total_power} proved {"mighty" if total_power > 100 else "adequate"} for this {"perilous" if danger_level == "High" else "challenging"} quest! """ print(adventure_story)

🏆 Step 6: Adventure Rating & Finale

# Final rating system (Lesson 6: Complex conditions) print("=" * 40) print("🏆 ADVENTURE RATING SYSTEM") print("=" * 40) # Calculate final score using multiple factors adventure_score = (total_power * 2) + experience_gained + (total_treasure_value // 10) # Determine rating with complex conditions if adventure_score >= 500: rating = "LEGENDARY HERO" emoji = "🌟" elif adventure_score >= 350: rating = "MASTER ADVENTURER" emoji = "🏆" elif adventure_score >= 200: rating = "SKILLED EXPLORER" emoji = "🎖️" elif adventure_score >= 100: rating = "BRAVE WANDERER" emoji = "⚔️" else: rating = "NOVICE ADVENTURER" emoji = "🗡️" # Display final results (Lesson 1: Printing + Lesson 4: String formatting) print(f"📊 Final Adventure Score: {adventure_score}") print(f"{emoji} Adventure Rating: {rating}") print() # Personalized message based on rating if adventure_score >= 400: message = f"Amazing work, {character_name}! You're destined for greatness!" elif adventure_score >= 200: message = f"Well done, {character_name}! You're becoming a true adventurer!" else: message = f"Great start, {character_name}! Every expert was once a beginner!" print(f"💌 {message}") # Fun facts using string operations and math print("\n" + "=" * 30) print("📊 FUN ADVENTURE FACTS") print("=" * 30) print(f"🔤 Your name has {len(character_name)} letters") print(f"🎨 Your favorite color '{favorite_color}' has {len(favorite_color)} letters") print(f"⚡ Your power is {total_power // 10} times stronger than average!") print(f"💰 You found {gold_found + gems_found} treasure items total!") print(f"📈 Adventure difficulty was {danger_level.upper()}!") print("\n🎉 Thanks for playing the Python Adventure Generator! 🎉")
