Lesson 14: Classes and Objects

🏗️ Learn Object-Oriented Programming to build organized, reusable code!

🏗️ What are Classes and Objects?

Classes are like blueprints for creating objects! Think of a class as a template that defines what properties and behaviors an object should have. An object is a specific instance created from that blueprint.

🎯 Object-Oriented Programming (OOP) Concepts

  • Class - A blueprint/template for creating objects
  • Object - A specific instance of a class
  • Attributes - Properties that store data about the object
  • Methods - Functions that define what the object can do
  • Inheritance - Creating new classes based on existing ones
  • Encapsulation - Keeping data and methods together

🏠 Real-World Analogy

Think of a House Blueprint:

  • 📐 Class (Blueprint): Defines rooms, windows, doors
  • 🏠 Object (Actual House): A specific house built from the blueprint
  • 🚪 Attributes: Number of rooms, color, address
  • 🔧 Methods: open_door(), turn_on_lights(), lock_house()

🎯 Step 1: Creating Your First Class

🐕 Example: Dog Class

class Dog: """A simple Dog class to demonstrate OOP concepts""" # Class attribute (shared by all dogs) species = "Canis lupus" def __init__(self, name, breed, age): """Constructor method - runs when creating a new dog""" self.name = name # Instance attribute self.breed = breed # Instance attribute self.age = age # Instance attribute self.energy = 100 # Default energy level def bark(self): """Method to make the dog bark""" print(f"{self.name} says: Woof! Woof!") def play(self): """Method to make the dog play""" if self.energy > 20: self.energy -= 20 print(f"{self.name} is playing! Energy: {self.energy}") else: print(f"{self.name} is too tired to play!") def sleep(self): """Method to restore the dog's energy""" self.energy = 100 print(f"{self.name} is sleeping... Energy restored!") def get_info(self): """Method to display dog information""" print(f"🐕 {self.name} - {self.breed}, {self.age} years old") # Creating objects (instances) from the Dog class my_dog = Dog("Buddy", "Golden Retriever", 3) friend_dog = Dog("Max", "German Shepherd", 5) # Using the objects my_dog.bark() my_dog.play() my_dog.get_info() print("\n" + "="*40 + "\n") friend_dog.bark() friend_dog.get_info()

🎯 Step 2: More Complex Example - Student Management System

class Student: """A comprehensive Student class""" # Class attribute to track total students total_students = 0 def __init__(self, name, age, student_id): """Initialize a new student""" self.name = name self.age = age self.student_id = student_id self.grades = [] # List to store grades self.courses = [] # List to store enrolled courses # Increment total students when a new student is created Student.total_students += 1 def add_grade(self, subject, grade): """Add a grade for a subject""" if 0 <= grade <= 100: self.grades.append({"subject": subject, "grade": grade}) print(f"✅ Added grade {grade} for {subject}") else: print("❌ Grade must be between 0 and 100!") def calculate_average(self): """Calculate the student's average grade""" if not self.grades: return 0 total = sum(grade_info["grade"] for grade_info in self.grades) return total / len(self.grades) def enroll_course(self, course_name): """Enroll in a new course""" if course_name not in self.courses: self.courses.append(course_name) print(f"📚 {self.name} enrolled in {course_name}") else: print(f"❌ Already enrolled in {course_name}") def get_grade_letter(self): """Convert average grade to letter grade""" avg = self.calculate_average() if avg >= 90: return "A" elif avg >= 80: return "B" elif avg >= 70: return "C" elif avg >= 60: return "D" else: return "F" def display_report_card(self): """Display a comprehensive report card""" print(f"\n📋 REPORT CARD FOR {self.name.upper()}") print("=" * 40) print(f"Student ID: {self.student_id}") print(f"Age: {self.age}") print(f"Courses: {', '.join(self.courses) if self.courses else 'None'}") print("\nGrades:") if self.grades: for grade_info in self.grades: print(f" {grade_info['subject']}: {grade_info['grade']}") print(f"\nAverage: {self.calculate_average():.1f}") print(f"Letter Grade: {self.get_grade_letter()}") else: print(" No grades recorded") print("=" * 40) @classmethod def get_total_students(cls): """Class method to get total number of students""" return cls.total_students def __str__(self): """String representation of the student""" return f"Student({self.name}, ID: {self.student_id})" # Example usage print("🏫 STUDENT MANAGEMENT SYSTEM") print("=" * 50) # Create students alice = Student("Alice Johnson", 16, "ST001") bob = Student("Bob Smith", 17, "ST002") charlie = Student("Charlie Brown", 16, "ST003") # Add courses and grades alice.enroll_course("Mathematics") alice.enroll_course("Physics") alice.enroll_course("Chemistry") alice.add_grade("Mathematics", 95) alice.add_grade("Physics", 87) alice.add_grade("Chemistry", 92) bob.enroll_course("Mathematics") bob.enroll_course("History") bob.add_grade("Mathematics", 78) bob.add_grade("History", 85) # Display report cards alice.display_report_card() bob.display_report_card() print(f"\n👥 Total Students: {Student.get_total_students()}")

🎯 Step 3: Inheritance - Creating Specialized Classes

# Base class class Vehicle: """Base class for all vehicles""" def __init__(self, make, model, year): self.make = make self.model = model self.year = year self.speed = 0 def start(self): print(f"🔑 {self.make} {self.model} is starting...") def stop(self): self.speed = 0 print(f"🛑 {self.make} {self.model} has stopped.") def accelerate(self, amount): self.speed += amount print(f"🚗 Speed increased to {self.speed} mph") def get_info(self): return f"{self.year} {self.make} {self.model}" # Derived class (inherits from Vehicle) class Car(Vehicle): """Car class that inherits from Vehicle""" def __init__(self, make, model, year, doors): super().__init__(make, model, year) # Call parent constructor self.doors = doors self.fuel = 100 def honk(self): print(f"🔊 {self.make} {self.model}: BEEP BEEP!") def refuel(self): self.fuel = 100 print(f"⛽ {self.make} {self.model} refueled!") def drive(self, distance): fuel_needed = distance * 0.1 if self.fuel >= fuel_needed: self.fuel -= fuel_needed print(f"🚗 Drove {distance} miles. Fuel remaining: {self.fuel:.1f}") else: print("❌ Not enough fuel!") # Another derived class class Motorcycle(Vehicle): """Motorcycle class that inherits from Vehicle""" def __init__(self, make, model, year, engine_size): super().__init__(make, model, year) self.engine_size = engine_size def wheelie(self): print(f"🏍️ {self.make} {self.model} is doing a wheelie!") def rev_engine(self): print(f"🏍️ {self.make} {self.model}: VROOOOM!") # Example usage print("🚗 VEHICLE INHERITANCE DEMO") print("=" * 40) # Create different vehicles my_car = Car("Toyota", "Camry", 2022, 4) my_bike = Motorcycle("Harley-Davidson", "Street 750", 2021, 750) # Use inherited methods my_car.start() my_car.accelerate(30) my_car.honk() my_car.drive(50) print(f"Car info: {my_car.get_info()}") print("\n" + "-" * 40 + "\n") my_bike.start() my_bike.accelerate(45) my_bike.rev_engine() my_bike.wheelie() print(f"Bike info: {my_bike.get_info()}")

💡 Key Concepts

🏗️ Class Definition

Create a blueprint for objects:

class MyClass: def __init__(self, value): self.value = value def my_method(self): return f"Value: {self.value}" # Create object obj = MyClass("Hello") print(obj.my_method())

🔧 Constructor Method

Initialize objects with __init__:

class Person: def __init__(self, name, age): self.name = name # Instance attribute self.age = age # Instance attribute def introduce(self): print(f"Hi, I'm {self.name}, {self.age} years old") person = Person("Alice", 25) person.introduce()

🧬 Inheritance

Create specialized classes from base classes:

class Animal: def __init__(self, name): self.name = name def speak(self): pass class Cat(Animal): def speak(self): return f"{self.name} says Meow!" cat = Cat("Whiskers") print(cat.speak())
