from datetime import datetime, timedelta from collections import defaultdict, Counter import bisect class Book: def __init__(self, isbn, title, author, genre, year): self.isbn = isbn self.title = title self.author = author self.genre = genre self.year = year self.is_checked_out = False self.checked_out_by = None self.due_date = None def __str__(self): status = "📕 Checked Out" if self.is_checked_out else "📗 Available" return f"{self.title} by {self.author} ({self.year}) - {status}" class Member: def __init__(self, member_id, name, email): self.member_id = member_id self.name = name self.email = email self.checked_out_books = [] self.reading_history = [] def __str__(self): return f"Member: {self.name} (ID: {self.member_id})" class Library: def __init__(self): self.books = {} # ISBN -> Book self.members = {} # ID -> Member self.books_by_title = [] # Sorted list for binary search self.books_by_author = defaultdict(list) self.books_by_genre = defaultdict(list) def add_book(self, book): self.books[book.isbn] = book bisect.insort(self.books_by_title, (book.title.lower(), book)) self.books_by_author[book.author].append(book) self.books_by_genre[book.genre].append(book) print(f"✅ Added: {book.title}") def binary_search_by_title(self, title): target = title.lower() left, right = 0, len(self.books_by_title) - 1 while left <= right: mid = (left + right) // 2 mid_title = self.books_by_title[mid][0] if mid_title == target: return self.books_by_title[mid][1] elif mid_title < target: left = mid + 1 else: right = mid - 1 return None
